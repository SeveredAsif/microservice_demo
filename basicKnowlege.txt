name: Build, Push, and Deploy

on:
  push:
    branches: [ main ]

jobs:
  build-and-deploy:
    runs-on: ubuntu-latest

    steps:
      # 1️⃣ Checkout your code
      - name: Checkout code
        uses: actions/checkout@v4

      # 2️⃣ Set up QEMU (for multi-arch builds)
      - name: Set up QEMU
        uses: docker/setup-qemu-action@v3

      # 3️⃣ Set up Python for running tests
      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: 3.11

      # 4️⃣ Install dependencies and run tests
      - name: Install dependencies
        run: pip install -r service1/requirements.txt

      - name: Run tests for Service1
        working-directory: ./service1
        run: pytest tests/

      # 5️⃣ Set up Docker Buildx
      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3

      # 6️⃣ Login to Docker Hub
      - name: Login to Docker Hub
        uses: docker/login-action@v3
        with:
          username: ${{ secrets.DOCKERHUB_USERNAME }}
          password: ${{ secrets.DOCKERHUB_TOKEN }}

      # 7️⃣ Build and push images
      - name: Build and push service1 image
        uses: docker/build-push-action@v5
        with:
          context: ./service1
          push: true
          tags: asifkhan131/service1:latest

      - name: Build and push service2 image
        uses: docker/build-push-action@v5
        with:
          context: ./service2
          push: true
          tags: asifkhan131/service2:latest

      # 8️⃣ Deploy to Azure VM over SSH
      - name: Deploy to Azure VM
        uses: appleboy/ssh-action@v0.1.9
        with:
          host: ${{ secrets.AZURE_VM_IP }}
          username: api_avengers
          key: ${{ secrets.AZURE_VM_KEY }}
          script: |
            # Stop and remove old containers
            docker stop service1 || true
            docker stop service2 || true
            docker rm service1 || true
            docker rm service2 || true

            # Remove old images
            docker rmi asifkhan131/service1:latest || true
            docker rmi asifkhan131/service2:latest || true

            # Pull latest images from Docker Hub
            docker pull asifkhan131/service1:latest
            docker pull asifkhan131/service2:latest

            # Run both containers
            docker network create micro_net
            docker run -d --name service2 --network micro_net -p 6001:6001 asifkhan131/service2:latest
            docker run -d --name service1 --network micro_net -p 80:6000 asifkhan131/service1:latest
